using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Учеба
{
    class House
    {
        public int Width { get; set; }
        public int Height { get; set; }

        public void ExpandSize(int w, int h)//изменение размеров
        {
            Width += w;
            Height += h;
        }

        public void ExpandAndGetSize(ref int w, ref int h)//передача значений, а не переменных
        {
            ExpandSize(w, h);//передача параметров на изменение размеров
            w = Width;
            h = Height;
        }
    }

    class Print
    {
        public void Show(string value)
        {
            Console.WriteLine(value);
        }
    }

    class Program
    {
        static void Main()
        {
            House house = new House();
            Print print = new Print();

            house.Width = 40;//устанавливаем значение
            house.Height = 30;

            int new_width = 4;//увеличиваем значение
            int new_height = 7;

            house.ExpandAndGetSize(ref new_width, ref new_height);//передача в метод ссылки на значение через ref
            print.Show($"Размеры {house.Width} и {house.Height}");//при обращении напрямую значение будут 44 и 37, а не 40 и 30
        }
    }
}
