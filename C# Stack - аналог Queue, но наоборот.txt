using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Data.SqlClient;
using System.Data.Common;
using System.Data;
using System.Collections;

namespace ADO
{
    class Program
    {
        static void Main(string[] args)
        {
            var item = new Stack();//Stack - аналог Queue, но делает все наоборот: работает с последним элементом списка
            item.Push("Hello");//Метод Push - добавляет элемент в список
            item.Push(123);
            item.Push('f');

            Console.WriteLine(item.Pop());//Pop - показывает ПОСЛЕДНИЙ элемент из коллекции и изымает его
            Console.WriteLine(new string('-', 30));

            foreach(var s in item)
            {
                Console.WriteLine(s.ToString());
            }
        }
    }
}
