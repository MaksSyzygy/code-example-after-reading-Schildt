using System;
using System.IO;

namespace Учеба
{
    class Program
    {
        static void Main(string[] args)
        {
            FileStream fout = null;

            try
            {
                fout = new FileStream(@"C:\Games\test.txt", FileMode.CreateNew);

                for(char c='A'; c <= 'Z'; c++)
                {
                    fout.WriteByte((byte)c);
                }
            }
            catch(IOException exc)
            {
                Console.WriteLine($"Ошибка ввода-вывода {exc.Message}");
            }
            finally
            {
                if (fout != null)
                {
                    fout.Close();
                }
            }
        }
    }
}
