using System;

namespace Учеба
{
    class Factorial
    {
        public int Fact(int n)
        {
            int t, result;

            result = 1;

            for (t = 1; t <= n; t++)
            {
                result *= t;
            }
            return result;
        }
    }

    class Program
    {
        static void Main()
        {
            Factorial factorial = new Factorial();
            for (int i = 1; i <= 10; i++)
            {
                Console.WriteLine($"Факториал числа {i} равен {factorial.Fact(i)}");
            }
        }
    }
}
