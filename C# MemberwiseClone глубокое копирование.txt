using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Data.SqlClient;
using System.Data.Common;
using System.Data;

namespace ADO
{
    class MyClass
    {
        public static string CompanyName = "У Сидоровича";
        public string Name;
        public int Age;
    }

    class Program : MyClass
    {
        static void Main(string[] args)
        {
            Program firstPerson = new Program();
            firstPerson.Age = 34;
            firstPerson.Name = "Петрович";

            Console.WriteLine($"Первый персонаж - {firstPerson.Name}, {firstPerson.Age} лет, {CompanyName}");

            Program secondPerson = firstPerson.MemberwiseClone() as Program;//Метод создает модифицированную копию создавая новый экзепляр на копии
            secondPerson.Name = "Иваныч";
            secondPerson.Age = 56;
            CompanyName = "У Иваныча";

            Console.WriteLine($"Второй персонаж - {secondPerson.Name}, {secondPerson.Age} лет, {CompanyName}");
        }
    }
}
