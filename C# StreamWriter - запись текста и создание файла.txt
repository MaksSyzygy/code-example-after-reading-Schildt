using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Data.SqlClient;
using System.Data.Common;
using System.Data;
using System.Collections;
using System.Collections.Specialized;
using System.IO;

namespace ADO
{
    class Program
    {
        static void Main(string[] args)
        {
            var file = new FileInfo(@"D:\Test\test.txt");//Создаем файл

            StreamWriter writer = file.CreateText();

            writer.WriteLine("Жили у бабуси ");//записываем текст
            writer.WriteLine("два веселых гуся");

            writer.Write(writer.NewLine);//переход на новую строку

            writer.WriteLine("Строка чисел:");

            for(int i = 0; i < 10; i++)
            {
                writer.Write(i + " ");//записываем цифры
            }

            writer.WriteLine(writer.NewLine);//новая строка

            writer.Close();//очищаем буферы

            Console.WriteLine("Файл создан и в него записан текст");
        }
    }
}
