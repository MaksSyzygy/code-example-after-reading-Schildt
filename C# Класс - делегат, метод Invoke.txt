using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Data.SqlClient;
using System.Data.Common;
using System.Data;

namespace ADO
{
    class Hello
    {
        public static void Greetings()
        {
            Console.WriteLine("Здарова, я - делегат!");
        }
    }

    public delegate void MyDelegate();

    class Program
    {
        static void Main(string[] args)
        {
            MyDelegate myDelegate = new MyDelegate(Hello.Greetings);//в тело класса-делегата должна быть передана цель!
            myDelegate();//вызов делегата
            myDelegate.Invoke();//либо вызов через метод Invoke
        }
    }
}
