using System;
using System.IO;

namespace Учеба
{
    class KeyEventArgs : EventArgs
    {
        public char ch;
    }

    class KeyEvent
    {
        public event EventHandler<KeyEventArgs> KeyPress;

        public void OnKeyPress(char key)
        {
            KeyEventArgs k = new KeyEventArgs();

            if (KeyPress != null)
            {
                k.ch = key;
                KeyPress(this, k);
            }
        }
    }
    class Program
    {
        static void Main()
        {
            KeyEvent keyEvent = new KeyEvent();

            ConsoleKeyInfo key;
            int count = 0;

            keyEvent.KeyPress += (sender, e) => Console.WriteLine($"Получено сообщение о нажатии клавиши {e.ch}");
            keyEvent.KeyPress += (sender, e) => count++;

            Console.WriteLine("Введите символы, по завершении точку");

            do
            {
                key = Console.ReadKey();
                keyEvent.OnKeyPress(key.KeyChar);
            } while (key.KeyChar != '.');

            Console.WriteLine($"Было нажато {count} клавиш");
        }
    }
}
