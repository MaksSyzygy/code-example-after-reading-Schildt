using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Data.SqlClient;
using System.Data.Common;
using System.Data;

namespace ADO
{
    class Index
    {
        private int[] num = new int[5];

        public int this[int index]//создаем индексатор, где this указывает на тип индексатора. У индексатора не может быть имени
        {
            get { return num[index]; }//возвращаем значения private num по индексу index
            set { num[index] = value; }//где num[index] принимает значение value, а value является значением
        }
    }
    class Program
    {
        static void Main(string[] args)
        {
            Index index = new Index();

            for(int i = 0; i < 5; i++)
            {
                index[i] = i;
                Console.WriteLine(index[i]);
            }
        }
    }
}
