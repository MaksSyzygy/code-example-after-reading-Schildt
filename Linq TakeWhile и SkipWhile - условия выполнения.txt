using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Data.SqlClient;
using System.Data.Common;
using System.Data;
using System.Collections.ObjectModel;//для подсключения Collection

namespace ADO
{
    class Program
    {
        static void Main(string[] args)
        {
            List<string> country = new List<string> { "Украина", "Голландия", "США", "Кюрасао", "Австралия", "Германия", "Швеция" };

            var TakeWhile = country.TakeWhile(x => x.Length > 3);//TakeWhile - пока lenght > 3 - элементы выводятся, после цикл прекращается

            foreach(var s in TakeWhile)
            {
                Console.WriteLine(s);
            }
            Console.WriteLine(new string('-', 30));

            var SkipWhile = country.SkipWhile(x => x.Length > 3);//SkipWhile - пропускает элементы до момента первого выполнения условия

            foreach(var s in SkipWhile)
            {
                Console.WriteLine(s);
            }
        }
    }
}
