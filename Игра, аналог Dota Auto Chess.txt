    class RandHero//класс в котором будет вычесляться случайный элемент массива
    {
        private static readonly Random rnd = new Random((int)DateTime.Now.Ticks);//текущее кол-во тиков реального времени

        public int Result(int lenght)//задаем констуктор для длины выбранного массива данных дабы не выйти за границы
        {
            Info info = new Info();

            int res = rnd.Next(0, lenght);//устанавливаем диапазон работы рандомайзера
            return res;
        }
    }

    class Rand//класс для вычесления
    {
        Info info = new Info();//инициализируем классы
        RandHero randHero = new RandHero();

        private static readonly Random rnd = new Random((int)DateTime.Now.Ticks);//аналогично, но для вычесления случайного числа для выбора случайного массива из героев

        public Rand()
        {
            for (int i = 0; i < 5; i++)//кол-во нужных героев
            {
                if (rnd.NextDouble() < 0.05)//вероятность выпадения героя по уровням
                {
                    Console.WriteLine(info.level_5[randHero.Result(info.level_5.Length)]);
                }
                else if (rnd.NextDouble() < 0.15)
                {
                    Console.WriteLine(info.level_4[randHero.Result(info.level_4.Length)]);
                }
                else if (rnd.NextDouble() < 0.25)
                {
                    Console.WriteLine(info.level_3[randHero.Result(info.level_3.Length)]);
                }
                else if (rnd.NextDouble() < 0.35)
                {
                    Console.WriteLine(info.level_2[randHero.Result(info.level_2.Length)]);
                }
                else
                {
                    Console.WriteLine(info.level_1[randHero.Result(info.level_1.Length)]);
                }
            }
        }
    }
