using System;

namespace Counter
{
    class CountDown
    {
        int val;

        public CountDown(int n)
        {
            val = n;
        }

        public int Count()
        {
            if (val > 0)
            {
                return val--;
            }
            else
            {
                return 0;
            }
        }
    }
}
