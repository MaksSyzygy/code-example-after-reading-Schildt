using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Data.SqlClient;
using System.Data.Common;
using System.Data;

namespace ADO
{
    class Program
    {
        static void Main(string[] args)
        {
            List<object> list = new List<object> { 1, 3, 5, 6, 34, 65 };

            foreach(var s in list.Cast<int>())//Cast - приводит тип к нужному типу данных. Если невозможно привести - будет вылет
            {
                Console.WriteLine(s);
            }
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Data.SqlClient;
using System.Data.Common;
using System.Data;

namespace ADO
{
    class Program
    {
        static void Main(string[] args)
        {
            List<object> list = new List<object> { 1, 3, 5, 6, 34, 65, "Hello", 123 };

            foreach(var s in list.OfType<int>())//OfType - если приведение невозможно - вылета не будет, а цикл пропустит неприводимое значение
            {
                Console.WriteLine(s);
            }
        }
    }
}
