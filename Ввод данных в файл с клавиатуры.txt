using System;
using System.IO;

namespace Учеба
{
    class Program
    {
        static void Main()
        {
            string str;
            FileStream fout;

            try
            {
                fout = new FileStream(@"C:\Games\text.txt", FileMode.Create);
            }
            catch(IOException exc)
            {
                Console.WriteLine($"Ошибка открытия файла {exc.Message}");
                return;
            }

            StreamWriter stream = new StreamWriter(fout);

            Console.WriteLine("Введите текст, а по окончании стоп");

            try
            {
                do
                {
                    Console.Write("Ваш текст: ");
                    str = Console.ReadLine();

                    if (str != "стоп")
                    {
                        str = str + "\r\n";
                        stream.Write(str);
                    }
                } while (str != "стоп");
            }
            catch(IOException exc)
            {
                Console.WriteLine($"Ошибка ввода-вывода: {exc.Message}");
            }
            finally
            {
                stream.Close();
            }
        }
    }
}
