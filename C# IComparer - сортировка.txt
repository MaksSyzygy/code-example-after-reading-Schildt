    public class DescendingComparer : IComparer
    {
        CaseInsensitiveComparer comparer = new CaseInsensitiveComparer();//сравнивает два объекта без учета регистра

        public int Compare(object x, object y)
        {
            int result = -1 * comparer.Compare(x, y);//сортировка по убыванию
            //int result = comparer.Compare(x, y);//сортировка по возрастанию
            return result;
        }
    }
